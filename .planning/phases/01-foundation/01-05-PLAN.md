    Create `test_tenant_isolation.py` — THE critical test for Phase 1:

    ```python
    # Test 1: Tenant A creates a user; Tenant B cannot see that user
    async def test_tenant_a_cannot_read_tenant_b_users(tenant_a_client, tenant_b_client, seed_two_tenants):
        # Create user as Tenant A
        resp = await tenant_a_client.post("/api/v1/users", json={"email": "alice@a.com"})
        assert resp.status_code == 201
        user_a_id = resp.json()["id"]

        # List users as Tenant B — should NOT include Tenant A's user
        resp = await tenant_b_client.get("/api/v1/users")
        assert resp.status_code == 200
        user_ids = [u["id"] for u in resp.json()]
        assert user_a_id not in user_ids, "Tenant B should not see Tenant A's users"

    # Test 2: Tenant B creates a user; Tenant A cannot see that user
    async def test_tenant_b_cannot_read_tenant_a_users(tenant_a_client, tenant_b_client, seed_two_tenants):
        # Mirror of test 1 with reversed tenants
        resp = await tenant_b_client.post("/api/v1/users", json={"email": "bob@b.com"})
        assert resp.status_code == 201
        user_b_id = resp.json()["id"]

        resp = await tenant_a_client.get("/api/v1/users")
        user_ids = [u["id"] for u in resp.json()]
        assert user_b_id not in user_ids

    # Test 3: Each tenant only sees their own users (positive case)
    async def test_each_tenant_sees_only_own_users(tenant_a_client, tenant_b_client, seed_two_tenants):
        await tenant_a_client.post("/api/v1/users", json={"email": "a1@a.com"})
        await tenant_a_client.post("/api/v1/users", json={"email": "a2@a.com"})
        await tenant_b_client.post("/api/v1/users", json={"email": "b1@b.com"})

        a_users = (await tenant_a_client.get("/api/v1/users")).json()
        b_users = (await tenant_b_client.get("/api/v1/users")).json()

        a_emails = {u["email"] for u in a_users}
        b_emails = {u["email"] for u in b_users}

        assert "a1@a.com" in a_emails
        assert "a2@a.com" in a_emails
        assert "b1@b.com" not in a_emails  # Cross-tenant leak check
        assert "b1@b.com" in b_emails
        assert "a1@a.com" not in b_emails  # Cross-tenant leak check

    # Test 4: No tenant header returns empty results (not error)
    async def test_no_tenant_header_returns_empty(async_client, seed_two_tenants):
        resp = await async_client.get("/api/v1/users")
        assert resp.status_code == 200
        assert len(resp.json()) == 0
    ```

    Create `test_role_endpoints.py`:
    ```python
    # Test: Assign all three role types to a user
    async def test_assign_all_role_types(tenant_a_client, seed_two_tenants):
        user = (await tenant_a_client.post("/api/v1/users", json={"email": "multi@a.com"})).json()
        for role in ["admin", "contractor", "client"]:
            resp = await tenant_a_client.post(
                f"/api/v1/users/{user['id']}/roles",
                json={"user_id": user["id"], "role": role}
            )
            assert resp.status_code == 201

        roles_resp = await tenant_a_client.get(f"/api/v1/users/{user['id']}/roles")
        roles = {r["role"] for r in roles_resp.json()}
        assert roles == {"admin", "contractor", "client"}

    # Test: Invalid role type is rejected
    async def test_invalid_role_rejected(tenant_a_client, seed_two_tenants):
        user = (await tenant_a_client.post("/api/v1/users", json={"email": "bad@a.com"})).json()
        resp = await tenant_a_client.post(
            f"/api/v1/users/{user['id']}/roles",
            json={"user_id": user["id"], "role": "superadmin"}
        )
        assert resp.status_code == 422  # Validation error
    ```
